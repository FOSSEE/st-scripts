\begin{frame}[fragile]
\frametitle{GCD}
\label{sec-4}

\lstset{language=Python}

\begin{footnotesize}
\begin{lstlisting}
def gcd(a, b):
    while b:
        a, b = b, a%b
    return a
    if gcd(40, 12) == 4:
        print "Everything OK"
    else:
        print "The GCD function is wrong"
\end{lstlisting}
\end{footnotesize}

\end{frame}
