\begin{frame}
\frametitle{Exercise 2}
\label{sec-4}


\begin{itemize}
\item What is the syntax to get the element `and' 
     in the list,listinlist ?
\item How would you get `and' using negative indices?
\end{itemize}
\end{frame}
\begin{frame}[fragile]
\frametitle{Solution 2}
\label{sec-5}

  
\lstset{language=Python}
\begin{lstlisting}

listinlist[1]
listinlist[-5]
\end{lstlisting}
\end{frame}
