\begin{frame}[fragile]
\frametitle{GCD}
\label{sec-4}

\lstset{language=Python}

\begin{footnotesize}
\begin{lstlisting}
def gcd(a, b):
    while b:
        a, b = b, a%b
    return a
if gcd(40, 12) == 4:
   print "Everything OK"
else:
   print "The GCD function is wrong"
\end{lstlisting}
\end{footnotesize}

\end{frame}

\label{sec-5}

\lstset{language=Python}
\begin{lstlisting}
if __name__ == "__main__":
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]
\frametitle{\_\_name\_\_ variable}
\begin{itemize}
\item The \_\_name\_\_ variable is local to every module.
\item It is equal to \_\_main\_\_ only when the file is run as a script.
\end{itemize}
\end{frame}
